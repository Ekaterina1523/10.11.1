<input type="text" id="fruit-color" placeholder="Цвет">

<input type="number" id="fruit-weight" placeholder="Вес">

<script>

    let fruits = [

        { kind: 'яблоко', color: 'красный', weight: 22 },

        { kind: 'банан', color: 'жёлтый', weight: 12 },

        { kind: 'груша', color: 'зелёный', weight: 47 },

        { kind: 'вишня', color: 'красный', weight: 5 },

        { kind: 'слива', color: 'фиолетовый', weight: 16 },

    ];


    function shuffle(array) {

        let shuffledArray = [...array];


        for (let i = shuffledArray.length - 1; i > 0; i--) {

            const j = Math.floor(Math.random() * (i + 1));

            [shuffledArray[i], shuffledArray[j]] = [shuffledArray[j], shuffledArray[i]];

        }


        if (JSON.stringify(shuffledArray) === JSON.stringify(array)) {

            alert('Порядок не изменился.');

        }


        return shuffledArray;

    }


    function filterByWeight(array, min, max) {

        return array.filter(fruit => fruit.weight >= min && fruit.weight <= max);

    }


    function bubbleSort(array, comparator) {

        let sortedArray = [...array];

        for (let i = 0; i < sortedArray.length; i++) {

            for (let j = 0; j < sortedArray.length - i - 1; j++) {

                if (comparator(sortedArray[j], sortedArray[j + 1]) > 0) {

                    [sortedArray[j], sortedArray[j + 1]] = [sortedArray[j + 1], sortedArray[j]];

                }

            }

        }

        return sortedArray;

    }


    function comparationColor(fruitA, fruitB) {

        const priority = ['красный', 'оранжевый', 'жёлтый', 'зелёный', 'синий', 'фиолетовый'];

        return priority.indexOf(fruitA.color) - priority.indexOf(fruitB.color);

    }


    function addFruit(kind, color, weight) {

        if (!kind || !color || !weight) {

            alert('Все поля должны быть заполнены!');

            return;

        }


        fruits.push({ kind, color, weight });

        renderFruits(fruits);

    }


    function renderFruits(list) {

        const ul = document.getElementById('fruit-list');

        ul.innerHTML = ''; // Очистить предыдущий список

        list.forEach((fruit, index) => {

            const li = document.createElement('li');

            li.textContent = `${index}: ${fruit.kind}, цвет: ${fruit.color}, вес: ${fruit.weight}`;

            ul.appendChild(li);

        });

    }


    document.getElementById('shuffle-btn').addEventListener('click', () => {

        fruits = shuffle(fruits);

        renderFruits(fruits);

    });


    document.getElementById('filter-btn').addEventListener('click', () => {

        const minWeight = parseInt(prompt('Введите минимальный вес:'), 10);

        const maxWeight = parseInt(prompt('Введите максимальный вес:'), 10);

        if (!isNaN(minWeight) && !isNaN(maxWeight)) {

            const filteredFruits = filterByWeight(fruits, minWeight, maxWeight);

            renderFruits(filteredFruits);

        }

    });


    document.getElementById('sort-btn').addEventListener('click', () => {

        const sortedFruits = bubbleSort(fruits, comparationColor);

        renderFruits(sortedFruits);

    });


    document.getElementById('add-fruit-btn').addEventListener('click', () => {

        const kind = document.getElementById('fruit-kind').value;

        const color = document.getElementById('fruit-color').value;

        const weight = parseInt(document.getElementById('fruit-weight').value, 10);

        addFruit(kind, color, weight);

    });


    renderFruits(fruits);

</script>
